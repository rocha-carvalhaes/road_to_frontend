sintaxe da condicional simples:


    if (condição) {
        'executa_caso_verdadeiro'
    } else {
        'executa_caso_falso'
    }

sintaxe da condicional múltipla:


    switch (expressao) { 
        case valor_1: 
            executa_caso_valor_1 
            break
        case valor_2:
            executa_caso_valor_2
            break
        case valor_3:
            executa_caso_valor_3
            break
        default:
    }
